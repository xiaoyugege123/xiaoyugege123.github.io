77. 组合

/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */
var combine = function (n, k) {
  let reult = [];
  let path = [];
  combineHelp(n, k, 1);
  return reult;
  // 结合树形图去写题目！
  function combineHelp(n, k, startIndex) {
    if (path.length == k) {
      reult.push([...path]);
      return;
    }
    for (let i = startIndex; i <= n; i++) {
      path.push(i);
      combineHelp(n, k, i + 1);
      path.pop();
    }
  }
};

216. 组合总和 III

/**
 * @param {number} k
 * @param {number} n
 * @return {number[][]}
 */
var combinationSum3 = function (k, n) {
  const result = [];
  const path = [];
  backtracking(n, k, 0, 1);
  return result;

  function backtracking(targetSum, k, sum, startIndex) {
    if (sum > targetSum) return; //剪枝操作
    if (path.length == k) {
      if (targetSum == sum) {
        result.push([...path]);
      }
      return;
    }
    //剪枝操作
    for (let i = startIndex; i <= n - (k - path.length) + 1; i++) {
      sum += i;
      path.push(i);
      backtracking(targetSum, k, sum, i + 1);
      sum -= i;
      path.pop();
    }
  }
};

39. 组合总和

/**
 * @param {number[]} candidates
 * @param {number} target
 * @return {number[][]}
 */
var combinationSum = function (candidates, target) {
  const result = [],
    path = [];
  backtracking(0, 0);
  return result;

  function backtracking(j, sum) {
    if (sum > target) return; //剪枝操作
    if (sum == target) {
      result.push([...path]);
      return;
    }
    for (let i = j; i < candidates.length; i++) {
      const num = candidates[i];
      sum += num;
      path.push(num);
      backtracking(i, sum);
      sum -= num;
      path.pop();
    }
  }
};

46. 全排列

/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var permute = function (nums) {
  const result = [],
    path = [];
  backtracking([]);
  return result;

  function backtracking(used) {
    if (path.length == nums.length) {
      result.push([...path]);
      return;
    }
    for (let i = 0; i < nums.length; i++) {
      if (used[i]) continue;
      path.push(nums[i]);
      used[i] = true;
      backtracking(used);
      path.pop();
      used[i] = false;
    }
  }
};

老师给的一道题

总体要求:传入二维数组，返回二维数组的所有排列组合
场景描述:
const arr = [
    ['黑色', '白色', '金色'],
    ['16G', '32G', '256G', '512G'],
    ['4.7', '5.5', '7.3'],
];
arr 数组包含了一个多维数组，
第一个子数组 ['黑色', '白色', '金色'] 包含了手机颜色的选项。
第二个子数组 ['16G', '32G', '256G', '512G'] 包含了手机存储容量的选项。
第三个子数组 ['4.7', '5.5', '7.3'] 包含了手机屏幕尺寸的选项。


编程输出所有可能的手机型号
[
  [ '黑色', '16G', '4.7' ],
  [ '白色', '16G', '4.7' ],
  [ '金色', '16G', '4.7' ],
  [ '黑色', '32G', '4.7' ],
  [ '白色', '32G', '4.7' ],
  [ '金色', '32G', '4.7' ],
  [ '黑色', '256G', '4.7' ],
  [ '白色', '256G', '4.7' ],
  ...
]

回溯如下：

function getAllCombinations(arrays) {
  //TODO:补充代码,建议用递归实现功能
  // 常规的回溯算法！！！
  let result = [];
  let temp = [];
  backtracking(0);
  return result;

  function backtracking(index) {
    if (index == arrays.length) {
      result.push([...temp]);
      return;
    }
    for (let i = 0; i < arrays[index].length; i++) {
      temp.push(arrays[index][i]);
      backtracking(index + 1);
      temp.pop();
    }
  }
}

22. 括号生成

/**
 * @param {number} n
 * @return {string[]}
 */
var generateParenthesis = function (n) {
  let result = [];
  function backTracking(s, open, close) {
    if (s.length == n * 2) {
      result.push(s);
      return;
    }
    if (open < n) {
      //还可以继续添加开括号
      backTracking(s + "(", open + 1, close);
    }
    if (close < open) {
      //还可以继续添加闭合括号
      backTracking(s + ")", open, close + 1);
    }
  }
  // 开始回溯
  backTracking("", 0, 0);
  return result;
};

79. 单词搜索

/**
 * @param {character[][]} board
 * @param {string} word
 * @return {boolean}
 */
var exist = function (board, word) {
  const words = word.split("");
  for (let i = 0; i < board.length; i++) {
    for (let j = 0; j < board[0].length; j++) {
      if (backTracking(board, words, i, j, 0)) return true;
    }
  }
  return false;
};

function backTracking(board, word, i, j, k) {
  //  (1) 行或列索引越界 或 (2) 当前矩阵元素与目标字符不同 或 (3) 当前矩阵元素已访问过 ； (3) 可合并至 (2)
  if (
    i >= board.length ||
    i < 0 ||
    j >= board[0].length ||
    j < 0 ||
    board[i][j] != word[k]
  )
    return false;
  // 字符串 word 已全部匹配
  if (k === word.length - 1) return true;
  board[i][j] = "*"; // 标记，进入内存递归
  const res =
    backTracking(board, word, i + 1, j, k + 1) ||
    backTracking(board, word, i - 1, j, k + 1) ||
    backTracking(board, word, i, j + 1, k + 1) ||
    backTracking(board, word, i, j - 1, k + 1);
  board[i][j] = word[k]; //还原当前矩阵值
  return res;
}

131. 分割回文串

/**
 * @param {string} s
 * @return {string[][]}
 */
const isPalindrome = (s, l, r) => {
  for (let i = l, j = r; i < j; i++, j--) {
    if (s[i] !== s[j]) return false;
  }
  return true;
};

var partition = function (s) {
  const res = [],
    path = [],
    len = s.length;
  backtracking(0);
  return res;
  function backtracking(i) {
    if (i >= len) {
      res.push(Array.from(path));
      return;
    }
    for (let j = i; j < len; j++) {
      if (!isPalindrome(s, i, j)) continue;
      path.push(s.slice(i, j + 1));
      backtracking(j + 1);
      path.pop();
    }
  }
};

51. N 皇后

/* 解题思路：
1. 以棋盘的左上角作为坐标原点，先遍历每一行，每一行放置一个皇后，并建一个path数组存储放置的坐标
2. 再遍历每一行的每一列，判断path数组中的坐标和当前的坐标会不会相互攻击；
    - 如果会相互攻击，则跳过本次循环；
    - 没有相互攻击，则将当前坐标放到path数组中（数组的索引为行，数组的值为列）
3. 最后，当遍历了所有的行结束，把path放到结果集中。
*/
var solveNQueens = function (n) {
  let res = [];
  backtrack(res, n, 0, []);
  return res;
};

/**
 * @param {* strng[][]} res 结果
 * @param {* number} n n 个皇后
 * @param {* number} rowInx 第几行的棋盘
 * @param {* number[]} path 皇后放置的位置
 */
const backtrack = (res, n, rowInx, path) => {
  if (rowInx === n) {
    // path 里的每一个数据的索引为行，值为皇后的列位置
    res.push(path.map((c) => ".".repeat(c) + "Q" + ".".repeat(n - 1 - c)));
    return;
  }

  for (let col = 0; col < n; col++) {
    // 是否能在当前位置放置皇后：
    //  a.是否在同一行： r === rowInx（因为遍历行，不会有重复，所以不需要这个条件）
    //  b.是否在同一列： c === col
    //  c.是否在正斜对角线： r+c === rowInx+col（「0,1」和「1,0」反斜对角相交）
    //  d.是否在反斜对角线： r-c === rowInx-col（「0,0」和「1,1」正斜对角相交）
    const canNotSet = path.some(
      (c, r) => c == col || r - c === rowInx - col || r + c === rowInx + col
    );
    if (!canNotSet) {
      path.push(col);
      backtrack(res, n, rowInx + 1, path);
      path.pop();
    }
  }
};

17. 电话号码的字母组合

/**
 * @param {string} digits
 * @return {string[]}
 */
var letterCombinations = function (digits) {
    const k = digits.length;
    const map = ["", "", "abc", "def", "ghi", "jkl", "mno", "pqrs", "tuv", "wxyz"];
    if (!k) return [];
    if (k === 1) return map[digits].split("");

    const res = [], path = [];
    backtracking(0);
    return res;

    function backtracking(index) {//这个a其实是一个index
        if (path.length === k) {
            res.push(path.join(""));
            return;
        }
        for (const v of map[digits[index]]) {
            path.push(v);
            backtracking(index + 1);
            path.pop();
        }
    }
};
